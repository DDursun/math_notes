\twocolumn[
\begin{center}
\textbf{\Large Ordinary Differential Equations (ODEs)}
\end{center}
\vspace{4pt}]

\textbf{0. Basics of Integration}

\textbf{Substitution:} $\int f(g(x))g'(x)\,dx = \int f(u)\,du$ where $u = g(x)$

\textbf{By Parts:} $\int u\,dv = uv - \int v\,du$

\textbf{LIATE Rule} (order for choosing $u$): Logs $\to$ Inverse trig $\to$ Algebraic $\to$ Trig $\to$ Exponential

\hrulefill

\textbf{1. Integrating Factor} -- Converts linear to exact.

Form: $\dfrac{dy}{dx} + P(x)y = Q(x)$, \quad $\mu(x) = e^{\int P(x)\,dx}$

Multiply all sides by $\mu(x)$, then $\frac{d}{dx}$ on left side.

\textbf{2. Exact Equations}

$M(x,y)\,dx + N(x,y)\,dy = 0$ is exact if $\dfrac{\partial M}{\partial y} = \dfrac{\partial N}{\partial x}$

To solve: (1) Integrate $M$ w.r.t.\ $x$, add $g(y)$. (2) Diff result w.r.t.\ $y$, set $= N(x,y)$. (3) Find $g'(y)$, integrate, write $\mu = C$.

\textbf{3. Check for Homogeneous}

$f(tx, ty) = t^n \cdot f(x,y) \Rightarrow$ degree $n$

\textbf{4. Homogeneous to Separable}

Sub $y = vx$: $\dfrac{dy}{dx} = v + x\dfrac{dv}{dx}$

Or $x = uy$: $\dfrac{dx}{dy} = u + y\dfrac{du}{dy}$

\textbf{5. Reduction of Order}

\textbf{First-Order System:} For $x'' + ax' + bx = 0$: let $x_1 = x$, $x_2 = x'$. Then $X' = AX$, $A = \begin{pmatrix} 0 & 1 \\ -b & -a \end{pmatrix}$, $X(t) = e^{At}X(0)$

\dotfill

\textbf{For 2nd solution} of $y'' + P(x)y' + Q(x)y = 0$: Given $y_1$, let $y_2 = v(x) \cdot y_1(x)$

$y_2' = v'y_1 + vy_1'$, \quad $y_2'' = v''y_1 + 2v'y_1' + vy_1''$. Plug in, simplify. \textbf{Tip:} $w = v'$

\textbf{6. Bernoulli Equations}

$\dfrac{dy}{dx} + P(x)y = Q(x)y^n$

(1) Divide by $y^n$. (2) Sub $v = y^{1-n}$, so $v' = (1-n)y^{-n}y'$. (3) Plug in. (4) Use IF($\mu$) to solve, plug back.

\textbf{7. Euler (Cauchy-Euler)}

$x^2y'' + axy' + by = 0$

Let $x = e^t$, $t = \ln x$:
\[
\frac{dy}{dx} = \frac{1}{x}\frac{dy}{dt}, \quad \frac{d^2y}{dx^2} = \frac{1}{x^2}\left(\frac{d^2y}{dt^2} - \frac{dy}{dt}\right)
\]

Or trial $y = x^m$: $xy' = mx^m$, $x^2y'' = m(m-1)x^m$

\hrulefill

\textbf{Characteristic Roots}

For $y'' + ay' + by = 0$, solution: $y = y_h + y_p$

\textbf{Case I} (distinct real): $y_h = C_1 e^{m_1 x} + C_2 e^{m_2 x}$

\textbf{Case II} (repeated): $y_h = (C_1 + C_2 x)e^{mx}$

\textbf{Case III} (complex $\alpha \pm i\beta$):
\[
y_h = e^{\alpha x}\left[C_1 \cos(\beta x) + C_2 \sin(\beta x)\right]
\]

Differential Operator: $D^2y = \dfrac{d^2y}{dx^2}$, \quad $Dy = \dfrac{dy}{dx}$

\hrulefill

\textbf{Undetermined Coefficients}

For RHS: $e^{ax}$, $\sin x$, $\cos x$, $x^n$

Steps: (1) Find $y_h$ from char.\ roots. (2) Guess $y_p$ from table. (3) Plug $y_p$ in ODE, solve for coeffs. (4) $y = y_h + y_p$

\begin{center}
\begin{tabular}{|c|c|}
\hline
\textbf{RHS} & \textbf{Guess $y_p$} \\
\hline
$ke^{ax}$ & $Ae^{ax}$ \\
\hline
$kx^n$ & $A_n x^n + \cdots + A_1 x + A_0$ \\
\hline
$\sin(bx)$ or $\cos(bx)$ & $A\cos(bx) + B\sin(bx)$ \\
\hline
$e^{ax}\sin(bx)$ & $e^{ax}[A\cos(bx) + B\sin(bx)]$ \\
\hline
$x^n e^{ax}$ & $(A_n x^n + \cdots + A_0)e^{ax}$ \\
\hline
$x\sin(bx)$ or $x\cos(bx)$ & $(Ax+B)\cos(bx) + (Cx+D)\sin(bx)$ \\
\hline
\end{tabular}
\end{center}

\textbf{Modification Rule ($n \geq 3$):} If $y_p \in y_h$, multiply by $x^s$ ($s$ = multiplicity of root)

\textbf{Operator Shift:} $P(D)[e^{ax}v(x)] = e^{ax}P(D+a)v(x)$

Ex: $P(D) = D^2(D-1)^3$, RHS $= e^x Ex^3$ ($a=1$). Replace $D \to (D+1)$: $P(D+1) = (D+1)^2 D^3$. Apply: $D^3(Ex^3) = 6E$, then $(D+1)^2(6E) = 6E$

\hrulefill

\textbf{Variation of Parameters}

$y'' + p(x)y' + q(x)y = g(x)$

Steps: (1) Find $y_h$ roots $\Rightarrow y_1, y_2$. (2) Compute Wronskian $W \neq 0$. (3) Solve for $u_1', u_2'$. (4) Integrate. (5) $y_p = u_1 y_1 + u_2 y_2$. (6) $y = C_1y_1 + C_2y_2 + y_p$

\textbf{2nd Order:}
\[
W = \begin{vmatrix} y_1 & y_2 \\ y_1' & y_2' \end{vmatrix} = y_1 y_2' - y_2 y_1'
\]
\[
u_1' = \frac{-y_2 \cdot g(x)}{W}, \quad u_2' = \frac{y_1 \cdot g(x)}{W}
\]

\textbf{General $n$th Order:}

$y^{(n)} + p_{n-1}y^{(n-1)} + \cdots + p_0 y = g(x)$

Given $n$ homogeneous solutions $y_1, y_2, \ldots, y_n$:
\[
W = \begin{vmatrix} 
y_1 & y_2 & \cdots & y_n \\
y_1' & y_2' & \cdots & y_n' \\
\vdots & \vdots & \ddots & \vdots \\
y_1^{(n-1)} & y_2^{(n-1)} & \cdots & y_n^{(n-1)}
\end{vmatrix}
\]

$y_p = \sum_{k=1}^{n} u_k(x) y_k(x)$ where $u_k' = \dfrac{W_k}{W}$

$W_k$ = $W$ with $k$th column replaced by $(0, 0, \ldots, 0, g(x))^T$

\textbf{3rd Order:}
\[
W = \begin{vmatrix} y_1 & y_2 & y_3 \\ y_1' & y_2' & y_3' \\ y_1'' & y_2'' & y_3'' \end{vmatrix}
\]

\[
u_1' = \frac{1}{W}\begin{vmatrix} 0 & y_2 & y_3 \\ 0 & y_2' & y_3' \\ g & y_2'' & y_3'' \end{vmatrix}, \quad
u_2' = \frac{1}{W}\begin{vmatrix} y_1 & 0 & y_3 \\ y_1' & 0 & y_3' \\ y_1'' & g & y_3'' \end{vmatrix}, \quad u_3' = \ldots
\]
