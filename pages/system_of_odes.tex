\begin{center}
\textbf{\Large Systems of ODEs and PDEs}
\end{center}
\vspace{4pt}
]
%----------------------------------------------
Many physics and engineering problems include coupled differential equations. For example, n coupled springs will result in n coupled differential equations arising from the equalization of Newtonian and Hooke's Forces. First-order ODE systems can also be used as a reduction of order method with the introduction of new variables for successive derivatives:

\[
y'' + 3y' + 2y = 0.
\]

Define the state variables
\[
x_1 = y, \qquad x_2 = y'.
\]

Then the system can be written as the following first-order system:
\[
\begin{bmatrix} x_1' \\ x_2' \end{bmatrix} = \begin{bmatrix} 0 & 1 \\ -2 & -3 \end{bmatrix} \begin{bmatrix} x_1 \\ x_2 \end{bmatrix}
\]
In general, $\mathbf{x}' = A\mathbf{x}$ where $A$ encodes the system coefficients.

%----------------------------------------------
\hrulefill

\textbf{Stability Classification (2D Systems)}

Based on eigenvalues $\lambda_1, \lambda_2$ of matrix $A$:

\begin{tabular}{@{}ll@{}}
\textbf{Stable node:} & both real, $\lambda_1, \lambda_2 < 0$ \\
\textbf{Unstable node:} & both real, $\lambda_1, \lambda_2 > 0$ \\
\textbf{Saddle:} & real, opposite signs \\
\textbf{Stable spiral:} & complex, $\text{Re}(\lambda) < 0$ \\
\textbf{Unstable spiral:} & complex, $\text{Re}(\lambda) > 0$ \\
\textbf{Center:} & purely imaginary $\lambda = \pm i\beta$ \\
\end{tabular}

\textbf{Quick check:} $\tau = a+d$ (trace), $\Delta = ad-bc$ (det)
\begin{itemize}\itemsep0pt
\item $\Delta < 0$: saddle (unstable)
\item $\Delta > 0$, $\tau < 0$: stable (node or spiral)
\item $\Delta > 0$, $\tau > 0$: unstable (node or spiral)
\item $\Delta > 0$, $\tau = 0$: center
\end{itemize}

\textbf{Solution to homogenous system}

\begin{enumerate}\itemsep0pt
\item \textbf{Real, distinct eigenvalues:}
\(\mathbf{x}(t)=c_1 e^{\lambda_1 t}\mathbf{v}_1+c_2 e^{\lambda_2 t}\mathbf{v}_2+\cdots\).

\item \textbf{Repeated real eigenvalue:}
\(\mathbf{x}(t)=c_1 e^{\lambda t}\mathbf{v}+c_2 e^{\lambda t}(t\mathbf{v}+\mathbf{u})\),
where \((A-\lambda I)\mathbf{u}=\mathbf{v}\).
If the multiplicity is three,
\[
\mathbf{x}(t)=c_1\mathbf{v}_1 e^{\lambda t}
+c_2(\mathbf{v}_1 t+\mathbf{v}_2)e^{\lambda t}
+c_3\!\left(\tfrac12\mathbf{v}_1 t^2+\mathbf{v}_2 t+\mathbf{v}_3\right)\!e^{\lambda t}.
\]

\item \textbf{Complex eigenvalues} \(\lambda=\alpha\pm i\beta\):
\(e^{\lambda t}=e^{\alpha t}(\cos\beta t+i\sin\beta t)\),
\(\mathbf{x}(t)=e^{\alpha t}(\mathbf{a}\cos\beta t-\mathbf{b}\sin\beta t)\).
\end{enumerate}

\textbf{Non-Homogeneous Systems}

For $\mathbf{Y}' = A\mathbf{Y} + \mathbf{F}(t)$ with $A$ upper triangular, solve bottom-up:

\textbf{Ex:} $\begin{bmatrix} y_1' \\ y_2' \\ y_3' \end{bmatrix} = \begin{bmatrix} 2 & 1 & 0 \\ 0 & 2 & 1 \\ 0 & 0 & 2 \end{bmatrix}\begin{bmatrix} y_1 \\ y_2 \\ y_3 \end{bmatrix} + \begin{bmatrix} e^{2t} \\ 3e^{2t} \\ 7e^{2t} \end{bmatrix}$

Expand: $y_3' = 2y_3 + 7e^{2t}$, \; $y_2' = 2y_2 + y_3 + 3e^{2t}$, \; $y_1' = 2y_1 + y_2 + e^{2t}$

\textbf{Step 1:} Solve $y_3' - 2y_3 = 7e^{2t}$ with $\mu = e^{-2t}$:
$\frac{d}{dt}(y_3 e^{-2t}) = 7$ $\Rightarrow$ $y_3 = (7t + C_3)e^{2t}$

\textbf{Step 2:} Substitute $y_3$ into $y_2' - 2y_2 = y_3 + 3e^{2t}$, solve with $\mu = e^{-2t}$

\textbf{Step 3:} Substitute $y_2$ into $y_1' - 2y_1 = y_2 + e^{2t}$, solve with $\mu = e^{-2t}$

\textbf{General formula:} $\mathbf{x}(t) = e^{At}C + e^{At}\displaystyle\int e^{-As}\mathbf{F}(s)\,ds$

%----------------------------------------------
\textbf{2nd Order Systems (Decoupling)}

For $\mathbf{x}'' = A\mathbf{x} + \mathbf{f}(t)$: Let $\mathbf{x} = P\mathbf{u}$ where $P$ = eigenvector matrix of $A$.

\textbf{Ex:} $\begin{bmatrix} x_1'' \\ x_2'' \end{bmatrix} = \begin{bmatrix} 3 & -2 \\ -2 & 3 \end{bmatrix}\begin{bmatrix} x_1 \\ x_2 \end{bmatrix} + \begin{bmatrix} f_1 \\ f_2 \end{bmatrix}$

Find eigenvalues $\lambda_1, \lambda_2$ and eigenvector matrix $P$. Substitute $\mathbf{x} = P\mathbf{u}$:

$P\mathbf{u}'' = AP\mathbf{u} + \mathbf{f}$ $\xrightarrow{\times P^{-1}}$ $\mathbf{u}'' = \underbrace{P^{-1}AP}_{D}\mathbf{u} + P^{-1}\mathbf{f}$

Since $D = \text{diag}(\lambda_1, \lambda_2)$, system decouples:
\[
u_1'' = \lambda_1 u_1 + (P^{-1}\mathbf{f})_1, \qquad u_2'' = \lambda_2 u_2 + (P^{-1}\mathbf{f})_2
\]
Solve each ODE by undetermined coefficients, then back-substitute $\mathbf{x} = P\mathbf{u}$.

%----------------------------------------------
\textbf{Matrix Exponentials $e^{At}$}

\textbf{Putzer's Method for (2$\times$2)}

$e^{At} = u_0 I + u_1 A$ for $2\times 2$ matrices

For distinct eigenvalues $\lambda_1, \lambda_2$:
\[
\begin{cases}
e^{\lambda_1 t} = u_0 + u_1\lambda_1 \\
e^{\lambda_2 t} = u_0 + u_1\lambda_2
\end{cases}
\]
Solving: $u_1 = \dfrac{e^{\lambda_1 t} - e^{\lambda_2 t}}{\lambda_1 - \lambda_2}$, \quad $u_0 = e^{\lambda_1 t} - u_1\lambda_1$

%----------------------------------------------
\textbf{Series Expansion} for $e^{At}$
\[
e^{At} = I + At + \frac{(At)^2}{2!} + \frac{(At)^3}{3!} + \cdots
\]
For nilpotent matrices ($A^k = 0$ for some $k$), series terminates.

%----------------------------------------------
\textbf{Diagonalization} for $e^{At}$

If $A = PDP^{-1}$ where $D = \text{diag}(\lambda_1, \lambda_2, \ldots)$ and $P$ has eigenvector columns:
\[
e^{At} = P e^{Dt} P^{-1}
\]
where $e^{Dt} = \text{diag}(e^{\lambda_1 t}, e^{\lambda_2 t}, \ldots)$
