%----------------------------------------------

\textbf{Real Distinct Eigenvalues:} $\mathbf{x}(t) = c_1 e^{\lambda_1 t}\mathbf{v}_1 + c_2 e^{\lambda_2 t}\mathbf{v}_2 + \cdots$

\textbf{Repeated Real Eigenvalue:} $\mathbf{x}(t) = c_1 e^{\lambda t}\mathbf{v} + c_2 e^{\lambda t}(t\mathbf{v} + \mathbf{u})$

where $(A - \lambda I)\mathbf{u} = \mathbf{v}$ (solve for $\mathbf{u}$ with $\mathbf{v}_1$ instead of $\mathbf{0}$)

If 3 repeated: $\mathbf{x}(t) = c_1 \mathbf{v}_1 e^{\lambda t} + c_2 (\mathbf{v}_1 t + \mathbf{v}_2) e^{\lambda t} + c_3 \left(\frac{1}{2}\mathbf{v}_1 t^2 + \mathbf{v}_2 t + \mathbf{v}_3\right) e^{\lambda t}$

\textbf{Complex Eigenvalues} $\lambda = \alpha \pm i\beta$:
\[
e^{\lambda t} = e^{(\alpha + i\beta)t} = e^{\alpha t}(\cos\beta t + i\sin\beta t)
\]
General solution (real form): $\mathbf{x}(t) = e^{\alpha t}\bigl(\mathbf{a}\cos\beta t - \mathbf{b}\sin\beta t\bigr)$

%----------------------------------------------
\textbf{Non-Homogeneous Systems}

For $\mathbf{Y}' = A\mathbf{Y} + \mathbf{F}(t)$ with $A$ upper triangular, solve bottom-up:

\textbf{Ex:} $\begin{bmatrix} y_1' \\ y_2' \\ y_3' \end{bmatrix} = \begin{bmatrix} 2 & 1 & 0 \\ 0 & 2 & 1 \\ 0 & 0 & 2 \end{bmatrix}\begin{bmatrix} y_1 \\ y_2 \\ y_3 \end{bmatrix} + \begin{bmatrix} e^{2t} \\ 3e^{2t} \\ 7e^{2t} \end{bmatrix}$

Expand: $y_3' = 2y_3 + 7e^{2t}$, \; $y_2' = 2y_2 + y_3 + 3e^{2t}$, \; $y_1' = 2y_1 + y_2 + e^{2t}$

\textbf{Step 1:} Solve $y_3' - 2y_3 = 7e^{2t}$ with $\mu = e^{-2t}$:
$\frac{d}{dt}(y_3 e^{-2t}) = 7$ $\Rightarrow$ $y_3 = (7t + C_3)e^{2t}$

\textbf{Step 2:} Substitute $y_3$ into $y_2' - 2y_2 = y_3 + 3e^{2t}$, solve with $\mu = e^{-2t}$

\textbf{Step 3:} Substitute $y_2$ into $y_1' - 2y_1 = y_2 + e^{2t}$, solve with $\mu = e^{-2t}$

\textbf{General formula:} $\mathbf{x}(t) = e^{At}C + e^{At}\displaystyle\int e^{-As}\mathbf{F}(s)\,ds$

%----------------------------------------------
\textbf{2nd Order Systems (Decoupling)}

For $\mathbf{x}'' = A\mathbf{x} + \mathbf{f}(t)$: Let $\mathbf{x} = P\mathbf{u}$ where $P$ = eigenvector matrix of $A$.

\textbf{Ex:} $\begin{bmatrix} x_1'' \\ x_2'' \end{bmatrix} = \begin{bmatrix} 3 & -2 \\ -2 & 3 \end{bmatrix}\begin{bmatrix} x_1 \\ x_2 \end{bmatrix} + \begin{bmatrix} f_1 \\ f_2 \end{bmatrix}$

Find eigenvalues $\lambda_1, \lambda_2$ and eigenvector matrix $P$. Substitute $\mathbf{x} = P\mathbf{u}$:

$P\mathbf{u}'' = AP\mathbf{u} + \mathbf{f}$ $\xrightarrow{\times P^{-1}}$ $\mathbf{u}'' = \underbrace{P^{-1}AP}_{D}\mathbf{u} + P^{-1}\mathbf{f}$

Since $D = \text{diag}(\lambda_1, \lambda_2)$, system decouples:
\[
u_1'' = \lambda_1 u_1 + (P^{-1}\mathbf{f})_1, \qquad u_2'' = \lambda_2 u_2 + (P^{-1}\mathbf{f})_2
\]
Solve each ODE by undetermined coefficients, then back-substitute $\mathbf{x} = P\mathbf{u}$.

%----------------------------------------------
\textbf{Putzer's Formula (2$\times$2)} for $e^{At}$

$e^{At} = u_0 I + u_1 A$ for $2\times 2$ matrices

For distinct eigenvalues $\lambda_1, \lambda_2$:
\[
\begin{cases}
e^{\lambda_1 t} = u_0 + u_1\lambda_1 \\
e^{\lambda_2 t} = u_0 + u_1\lambda_2
\end{cases}
\]
Solving: $u_1 = \dfrac{e^{\lambda_1 t} - e^{\lambda_2 t}}{\lambda_1 - \lambda_2}$, \quad $u_0 = e^{\lambda_1 t} - u_1\lambda_1$

%----------------------------------------------
\textbf{Series Expansion} for $e^{At}$
\[
e^{At} = I + At + \frac{(At)^2}{2!} + \frac{(At)^3}{3!} + \cdots
\]
For nilpotent matrices ($A^k = 0$ for some $k$), series terminates.

%----------------------------------------------
\textbf{Diagonalization} for $e^{At}$

If $A = PDP^{-1}$ where $D = \text{diag}(\lambda_1, \lambda_2, \ldots)$ and $P$ has eigenvector columns:
\[
e^{At} = P e^{Dt} P^{-1}
\]
where $e^{Dt} = \text{diag}(e^{\lambda_1 t}, e^{\lambda_2 t}, \ldots)$

\textbf{Laplace Transform for PDEs}

Transform time derivatives (treating $x$ as parameter):

$\mathcal{L}\{u_t\} = sU - u(x,0)$

$\mathcal{L}\{u_{tt}\} = s^2U - su(x,0) - u_t(x,0)$

$\mathcal{L}\{u_x\} = U_x$ \quad ($s$ is just a constant!)

\textbf{Result:} PDE $\to$ ODE in $x$ with parameter $s$

\textbf{Solve ODE using:}
\begin{itemize}
\item 1st order: integrating factor $\mu$
\item 2nd order: homogeneous solution or undetermined coefficients
\end{itemize}

\textbf{Boundary conditions:} If $u(0,t) = g(t)$, then $\mathcal{L}\{u(0,t)\} = U(0,s)$

%----------------------------------------------
\textbf{Separation of Variables}

Assume $u = X(x)Y(y)$ (or $u = X(x)T(t)$, etc.)

Partial derivatives become: $u_x = X'Y$ where $X' = \dfrac{dX}{dx}$, \quad $u_y = XY'$

Substitute into PDE $\to$ separate variables $\to$ each side equals constant $-\lambda$

%----------------------------------------------
\textbf{3D Separation (Two Spatial Dims)}

For equation with 3 variables and 2 independent summation indices $m$ and $n$:
\[
u(x,y,t) = \sum_{m=1}^{\infty}\sum_{n=1}^{\infty} A_{mn}\sin\!\left(\frac{m\pi x}{D}\right)\sin\!\left(\frac{n\pi y}{C}\right)
\]
(Coefficients $A_{mn}$ may include time-dependent factor)

%----------------------------------------------
\textbf{Polynomial Long Division}

Divide leading terms $\to$ multiply divisor by result $\to$ subtract $\to$ bring down $\to$ repeat.
